59 @router.get("/callback/")
60 async def callback(request: Request, osm_auth=Depends(init_osm_auth)):
61  """Performs oauth token exchange with OpenStreetMap.
62 
63  Provides an access token that can be used for authenticating other endpoints.
64  Also returns a cookie containing the access token for persistence in frontend apps.
65 
66  Args:
67  request: The GET request.
68  request: The response, including a cookie.
69  osm_auth: The Auth object from osm-login-python.
70 
71  Returns:
72  access_token (string): The access token provided by the login URL request.
73  """
74  log.debug(f"Callback url requested: {request.url}")
75 
76  # Enforce https callback url for openstreetmap.org
77  callback_url = str(request.url).replace("http://", "https://")
78 
79  # Get access token
80  access_token = osm_auth.callback(callback_url).get("access_token")
81  log.debug(f"Access token returned of length {len(access_token)}")
82  response = JSONResponse(content={"access_token": access_token}, status_code=200)
83 
84  # Set cookie
85  cookie_name = settings.FMTM_DOMAIN.replace(".", "_")
86  log.debug(
87  f"Setting cookie in response named '{cookie_name}' with params: "
88  f"max_age=259200 | expires=259200 | path='/' | "
89  f"domain={settings.FMTM_DOMAIN} | httponly=True | samesite='lax' | "
90  f"secure={False if settings.DEBUG else True}"
91  )
92  response.set_cookie(
93  key=cookie_name,
94  value=access_token,
95  max_age=31536000, # OSM currently has no expiry
96  expires=31536000, # OSM currently has no expiry,
97  path="/",
98  domain=settings.FMTM_DOMAIN,
99  secure=False if settings.DEBUG else True,
100  httponly=True,
101  samesite="lax",
102  )
103  return response

131  """Get user from User table if exists, else create."""
132  existing_user = await user_crud.get_user(db, user_data.id)
133 
134  if existing_user:
135  # Update an existing user
136  if user_data.img_url:
137  existing_user.profile_img = user_data.img_url
138  db.commit()
139  return existing_user
140 
141  user_by_username = await user_crud.get_user_by_username(db, user_data.username)
142  if user_by_username:
143  raise HTTPException(
144  status_code=400,
145  detail=(
146  f"User with this username {user_data.username} already exists. "
147  "Please contact the administrator."
148  ),
149  )
150 
151  # Add user to database
152  db_user = DbUser(
153  id=user_data.id,
154  username=user_data.username,
155  profile_img=user_data.img_url,
156  role=user_data.role,
157  )
158  db.add(db_user)
159  db.commit()
160 
161  return db_user
