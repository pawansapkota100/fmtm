52 async def check_access(
53  user: Union[AuthUser, int],
54  db: Session,
55  org_id: Optional[int] = None,
56  project_id: Optional[int] = None,
57  role: Optional[ProjectRole] = None,
58 ) -> Optional[DbUser]:
59  """Check if the user has access to a project or organisation.
60 
61  Access is determined based on the user's role and permissions:
62  - If the user has an 'ADMIN' role, access is granted.
63  - If the user has a 'READ_ONLY' role, access is denied.
64  - For other roles, access is granted if the user is an organisation manager
65  for the specified organisation (org_id) or has the specified role
66  in the specified project (project_id).
67 
68  Args:
69  user (AuthUser, int): AuthUser object, or user ID.
70  db (Session): SQLAlchemy database session.
71  org_id (Optional[int]): Org ID for organisation-specific access.
72  project_id (Optional[int]): Project ID for project-specific access.
73  role (Optional[ProjectRole]): Role to check for project-specific access.
74 
75  Returns:
76  Optional[DbUser]: The user details if access is granted, otherwise None.
77  """
78  user_id = user if isinstance(user, int) else await get_uid(user)
79 
80  sql = text(
81  """
82  SELECT *
83  FROM users
84  WHERE id = :user_id
85  AND (
86  CASE
87  WHEN role = 'ADMIN' THEN true
88  WHEN role = 'READ_ONLY' THEN false
89  ELSE
90  EXISTS (
91  SELECT 1
92  FROM organisation_managers
93  WHERE organisation_managers.user_id = :user_id
94  AND organisation_managers.organisation_id = :org_id
95  )
96  OR EXISTS (
97  SELECT 1
98  FROM user_roles
99  WHERE user_roles.user_id = :user_id
100  AND user_roles.project_id = :project_id
101  AND user_roles.role >= :role
102  )
103  END
104  );
105  """
106  )
107 
108  result = db.execute(
109  sql,
110  {
111  "user_id": user_id,
112  "project_id": project_id,
113  "org_id": org_id,
114  "role": getattr(role, "name", None),
115  },
116  )
117  db_user = result.first()
118 
119  if db_user:
120  return DbUser(**db_user._asdict())
121 
122  return None

175 async def org_admin(
176  project: Optional[DbProject] = Depends(get_project_by_id),
177  org_id: Optional[int] = None,
178  db: Session = Depends(get_db),
179  user_data: AuthUser = Depends(login_required),
180 ) -> dict:
181  """Organisation admin with full permission for projects in an organisation.
182 
183  Returns:
184  dict: in format {'user': DbUser, 'org': DbOrganisation}.
185  """
186  if not (project or org_id):
187  raise HTTPException(
188  status_code=HTTPStatus.BAD_REQUEST,
189  detail="Either org_id or project_id must be provided",
190  )
191 
192  if project and org_id:
193  log.error("Both org_id and project_id cannot be passed at the same time")
194  raise HTTPException(
195  status_code=HTTPStatus.BAD_REQUEST,
196  detail="Both org_id and project_id cannot be passed at the same time",
197  )
198 
199  # Extract org id from project if passed
200  if project:
201  org_id = project.organisation_id
202 
203  # Ensure org_id is provided, raise an exception otherwise
204  if not org_id:
205  raise HTTPException(
206  status_code=HTTPStatus.BAD_REQUEST,
207  detail="org_id must be provided to check organisation admin role",
208  )
209 
210  org_user_dict = await check_org_admin(
211  db,
212  user_data,
213  org_id=org_id,
214  )
215 
216  if project:
217  org_user_dict["project"] = project
218 
219  return org_user_dict
