101 async def update_task_status(
102  db: Session, user_id: int, task_id: int, new_status: TaskStatus
103 ):
104  """Update the status of a task."""
105  log.debug(f"Updating task ID {task_id} to status {new_status}")
106  if not user_id:
107  log.error(f"User id is not present: {user_id}")
108  raise HTTPException(status_code=400, detail="User id required.")
109 
110  db_user = await user_crud.get_user(db, user_id)
111  if not db_user:
112  msg = f"User with id {user_id} does not exist."
113  log.error(msg)
114  raise HTTPException(status_code=400, detail=msg)
115 
116  db_task = await get_task(db, task_id)
117  log.debug(f"Returned task from db: {db_task}")
118 
119  if db_task:
120  if db_task.task_status in [
121  TaskStatus.LOCKED_FOR_MAPPING,
122  TaskStatus.LOCKED_FOR_VALIDATION,
123  ]:
124  log.debug(f"Task {task_id} currently locked")
125  if not (user_id is not db_task.locked_by):
126  msg = (
127  f"User {user_id} with username {db_user.username} "
128  "has not locked this task."
129  )
130  log.error(msg)
131  raise HTTPException(
132  status_code=403,
133  detail=msg,
134  )
135 
136  if not verify_valid_status_update(db_task.task_status, new_status):
137  msg = f"{new_status} is not a valid task status"
138  log.error(msg)
139  raise HTTPException(
140  status_code=422,
141  detail=msg,
142  )
143 
144  # update history prior to updating task
145  update_history = await create_task_history_for_status_change(
146  db_task, new_status, db_user
147  )
148  db.add(update_history)
149 
150  db_task.task_status = new_status
151 
152  if new_status in [
153  TaskStatus.LOCKED_FOR_MAPPING,
154  TaskStatus.LOCKED_FOR_VALIDATION,
155  ]:
156  db_task.locked_by = db_user.id
157  else:
158  db_task.locked_by = None
159 
160  if new_status == TaskStatus.MAPPED:
161  db_task.mapped_by = db_user.id
162  if new_status == TaskStatus.VALIDATED:
163  db_task.validated_by = db_user.id
164  if new_status == TaskStatus.INVALIDATED:
165  db_task.mapped_by = None
166 
167  db.commit()
168  db.refresh(db_task)
169  return db_task
170 
171  else:
172  raise HTTPException(
173  status_code=400,
174  detail=(
175  f"Not a valid status update: "
176  f"{db_task.task_status.name} to {new_status.name}"
177  ),
178  )

225 async def update_task_files(
226  db: Session,
227  project_id: int,
228  project_odk_id: int,
229  project_name: str,
230  task_id: int,
231  category: str,
232  task_boundary: str,
233 ):
234  """Update associated files for a task."""
235  # This file will store osm extracts
236  task_polygons = f"/tmp/{project_name}_{category}_{task_id}.geojson"
237 
238  # Update data extracts in the odk central
239  pg = PostgresClient("underpass")
240 
241  category = "buildings"
242 
243  # This file will store osm extracts
244  outfile = f"/tmp/test_project_{category}.geojson"
245 
246  # Delete all tasks of the project if there are some
247  db.query(db_models.DbFeatures).filter(
248  db_models.DbFeatures.task_id == task_id
249  ).delete()
250 
251  # OSM Extracts
252  outline_geojson = pg.getFeatures(
253  boundary=task_boundary,
254  filespec=outfile,
255  polygon=True,
256  xlsfile=f"{category}.xls",
257  category=category,
258  )
259 
260  updated_outline_geojson = {"type": "FeatureCollection", "features": []}
261 
262  # Collect feature mappings for bulk insert
263  for feature in outline_geojson["features"]:
264  # If the osm extracts contents do not have a title,
265  # provide an empty text for that
266  feature["properties"]["title"] = ""
267 
268  feature_shape = shape(feature["geometry"])
269 
270  wkb_element = from_shape(feature_shape, srid=4326)
271  updated_outline_geojson["features"].append(feature)
272 
273  db_feature = db_models.DbFeatures(
274  project_id=project_id,
275  geometry=wkb_element,
276  properties=feature["properties"],
277  )
278  db.add(db_feature)
279  db.commit()
280 
281  # Update task_polygons file containing osm extracts with the new
282  # geojson contents containing title in the properties.
283  with open(task_polygons, "w") as jsonfile:
284  jsonfile.truncate(0) # clear the contents of the file
285  dump(updated_outline_geojson, jsonfile)
286 
287  # Update the osm extracts in the form.
288  central_crud.upload_xform_media(project_odk_id, task_id, task_polygons, None)
289 
290  return True

468 async def count_validated_and_mapped_tasks(
469  task_history: list[db_models.DbTaskHistory], end_date: datetime
470 ) -> list[tasks_schemas.TaskHistoryCount]:
471  """Counts the number of validated and mapped tasks.
472 
473  Args:
474  task_history: The task history records to count.
475  end_date: The end date of the date range.
476 
477  Returns:
478  A list of dictionaries with following keys:
479  - 'date': The date in the format 'MM/DD'.
480  - 'validated': The cumulative count of validated tasks.
481  - 'mapped': The cumulative count of mapped tasks.
482  """
483  cumulative_counts = {}
484  results = []
485 
486  current_date = end_date
487  while current_date <= datetime.now():
488  date_str = current_date.strftime("%m/%d")
489  cumulative_counts = {"date": date_str, "validated": 0, "mapped": 0}
490  results.append(cumulative_counts)
491  current_date += timedelta(days=1)
492 
493  # Populate cumulative_counts with counts from task_history
494  for result in task_history:
495  task_status = result.action_text.split()[5]
496  date_str = result.action_date.strftime("%m/%d")
497  entry = next((entry for entry in results if entry["date"] == date_str), None)
498 
499  if entry:
500  if task_status == "VALIDATED":
501  entry["validated"] += 1
502  elif task_status == "MAPPED":
503  entry["mapped"] += 1
504 
505  total_validated = 0
506  total_mapped = 0
507 
508  for entry in results:
509  total_validated += entry["validated"]
510  total_mapped += entry["mapped"]
511  entry.update({"validated": total_validated, "mapped": total_mapped})
512 
513  return results
