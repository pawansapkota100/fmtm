161 def verify_valid_status_update(old_status: TaskStatus, new_status: TaskStatus):
162  """Verify the status update is valid, inferred from previous state."""
163  if old_status is TaskStatus.READY:
164  return new_status in [
165  TaskStatus.LOCKED_FOR_MAPPING,
166  TaskStatus.BAD,
167  TaskStatus.SPLIT,
168  ]
169  elif old_status is TaskStatus.LOCKED_FOR_MAPPING:
170  return new_status in [
171  TaskStatus.READY,
172  TaskStatus.MAPPED,
173  TaskStatus.BAD,
174  TaskStatus.SPLIT,
175  ]
176  elif old_status is TaskStatus.MAPPED:
177  return new_status in [
178  TaskStatus.LOCKED_FOR_MAPPING,
179  TaskStatus.LOCKED_FOR_VALIDATION,
180  ]
181  elif old_status is TaskStatus.LOCKED_FOR_VALIDATION:
182  return new_status in [TaskStatus.INVALIDATED, TaskStatus.VALIDATED]
183  elif old_status is TaskStatus.VALIDATED:
184  return new_status == TaskStatus.INVALIDATED
185  elif old_status is TaskStatus.INVALIDATED:
186  return new_status in [
187  TaskStatus.LOCKED_FOR_MAPPING,
188  TaskStatus.BAD,
189  TaskStatus.SPLIT,
190  ]
191  elif old_status is TaskStatus.BAD:
192  return new_status == TaskStatus.ARCHIVED
193  elif old_status is TaskStatus.SPLIT:
194  return new_status == TaskStatus.ARCHIVED
