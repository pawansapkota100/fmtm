620  log.debug("Validating uploaded XLS form")
621

623  file_ext = file_path.suffix.lower()
624  allowed_extensions = {".xls", ".xlsx", ".xml"}
625  if file_ext not in allowed_extensions:
626  raise HTTPException(

648  project_id,
649  BytesIO(custom_xls_form) if custom_xls_form else None,
650  form_category,
651  file_ext if xls_form_upload else "xls",
652  background_task_id,
