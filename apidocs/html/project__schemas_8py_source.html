343  @field_serializer("last_active")
344  def get_last_active(self, value, values):
345  """Date of last activity on project."""
346  if value is None:
347  return None
348 
349  last_active = parser.parse(value).replace(tzinfo=None)
350  current_date = datetime.now()
351 
352  time_difference = current_date - last_active
353 
354  days_difference = time_difference.days
355 
356  if days_difference == 0:
357  return "today"
358  elif days_difference == 1:
359  return "yesterday"
360  elif days_difference < 7:
361  return f'{days_difference} day{"s" if days_difference > 1 else ""} ago'
362  else:
363  return last_active.strftime("%d %b %Y")
