51 @router.get("/{id}", response_model=user_schemas.UserOut)
52 async def get_user_by_identifier(id: str, db: Session = Depends(database.get_db)):
53  """Get a single users details.
54 
55  The OSM ID should be used.
56  If this is not known, the endpoint falls back to searching
57  for the username.
58  """

61  try:

63  user = await user_crud.get_user(db, user_id=osm_id)
64  except ValueError:
65  # Skip if not a valid integer
66  pass
67 
68  if not user:
69  # Search by Username
70  user = await user_crud.get_user_by_username(db, username=id)
71 
72  # No user found
73  if not user:
74  raise HTTPException(status_code=404, detail="User not found")
75 
76  return user
